💻 Ejemplo 4: Uso del Componente

HTML

<button class="producto-button" id="mi-producto"></button>

JavaScript

// Obtener elemento const element = document.getElementById('mi-producto'); // Inicializar componente const productoBtn = new ProductoButton(element, { id: 'pizza-margarita', nombre: 'Margarita', precio: 8.50, categoria: 'pizzas', evento: 'oferta' }); // Eventos productoBtn.on('producto:add', (data) => { console.log('Añadido:', data); // Añadir a la orden... }); productoBtn.on('producto:variations', (data) => { console.log('Ver variaciones:', data); // Mostrar modal de variaciones... }); // Actualizar precio productoBtn.update({ precio: 7.50 }); // Cambiar evento productoBtn.updateEvento('destacado');
