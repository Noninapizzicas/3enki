💻 Ejemplo 4: Uso del Componente

HTML

<button class="cuenta-button" data-id="mesa-1" data-tipo="local" data-estado="listo"></button>

JavaScript

// Obtener el elemento const element = document.querySelector('.cuenta-button'); // Inicializar componente const cuentaBtn = new CuentaButton(element, { id: 'mesa-1', nombre: 'Mesa 1', tipo: 'local', estado: 'listo', total: 125.50, tiempo: 15, config: { apiBaseUrl: 'http://localhost:3000/api', enableMQTT: true } }); // Eventos cuentaBtn.on('action-comandero', (data) => { console.log('Abrir comandero:', data); }); cuentaBtn.on('action-cobro', (data) => { console.log('Procesar cobro:', data); }); cuentaBtn.on('state-change', (data) => { console.log('Estado cambió:', data); }); // Actualizar estado await cuentaBtn.updateEstado('pagado'); // Actualizar datos cuentaBtn.update({ total: 150.00, tiempo: 20 }); // Destruir componente cuentaBtn.destroy();
